(1 + 1 == 0) or False   # False, neither statement is True

(2 < 0) or True         # True

(2 > 0) and False         # False, using "and" if any statement is False, the evaluation is false

(3 == 8) or (3 > 4)     # False

13 - 6 != (3 * 2) + 1 # False, 7 IS equal to 7 

13 - 6 == (3 * 2) + 1 # True

(2 + 2 + 2 >= 6) and (-1 * -1 < 0) # False

(2 + 2 + 2 >= 6) or (-1 * -1 < 0) # True

(4 * 2 <= 8) and (7 - 1 == 6) # True

not True == False # True

not False == True # True

not True == True # False

not False == False # False

not (True == False or True == True) # False
